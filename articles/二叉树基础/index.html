/**
 * @param {TreeNode} root
 * @param {number} sum
 * @return {boolean}
 */
var hasPathSum = function(root, sum) {
    return helper(root, sum)
};

function helper(node, sum){
    if(!node){
        return false
    }else{
        sum -= node.val
    }
    if(!node.left && !node.right){
        return sum === 0
    }
    return helper(node.left, sum) || helper(node.right, sum)
}

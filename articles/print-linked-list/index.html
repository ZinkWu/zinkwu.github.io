 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {number[]}
 */
var reversePrint = function(head) {
    if(head === null) return []
   let result = []
   while(head){
       result.unshift(head.val)
       head = head.next
   }
   return result
};

function reverseList(head){
    let next;
    let pre;
    while(head){
        next = head.next;
        head.next = pre;
        pre = head;
        head = next;
    }
    return pre
}
